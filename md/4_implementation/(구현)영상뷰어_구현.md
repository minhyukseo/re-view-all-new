### [Backend] server/src/index.ts (영상 추출 API 추가)
```typescript
// 1. 영상 추출 전용 엔드포인트 추가
app.get('/api/video-extract', async (c) => {
  const url = c.req.query('url');
  if (!url) return c.json({ error: 'URL is required' }, 400);

  try {
    const { html } = await fetchText(url, c.env);
    const $ = load(html);
    
    // 유튜브 체크
    if (url.includes('youtube.com') || url.includes('youtu.be')) {
      const videoId = url.includes('v=') ? url.split('v=')[1].split('&')[0] : url.split('/').pop();
      return c.json({
        type: 'youtube',
        src: `https://www.youtube.com/embed/${videoId}`,
        title: $('title').text().trim() || 'YouTube Video'
      });
    }

    // 스트리머블 체크
    if (url.includes('streamable.com')) {
      const videoId = url.split('/').pop();
      return c.json({
        type: 'streamable',
        src: `https://streamable.com/e/${videoId}`,
        title: $('title').text().trim() || 'Streamable Video'
      });
    }

    // 일반 사이트 영상 소스 추출 (og:video 또는 <video> 태그)
    const directVideo = 
      $('meta[property="og:video"]').attr('content') || 
      $('meta[property="og:video:url"]').attr('content') ||
      $('video source').first().attr('src') ||
      $('video').first().attr('src');

    if (directVideo) {
      return c.json({
        type: 'direct',
        src: normalizeUrl(directVideo, url),
        title: $('meta[property="og:title"]').attr('content') || $('title').text().trim() || 'Video Content'
      });
    }

    // 최후의 수단: iframe 찾기
    const iframeSrc = $('iframe').first().attr('src');
    if (iframeSrc) {
      return c.json({
        type: 'embed',
        src: normalizeUrl(iframeSrc, url),
        title: $('title').text().trim() || 'Embedded Video'
      });
    }

    return c.json({ error: 'No video content found' }, 404);
  } catch (err) {
    return c.json({ error: 'Extraction failed' }, 500);
  }
});
```

### [Frontend] web/components/VideoPlayer.tsx
```tsx
"use client";

import React from "react";
import { motion } from "framer-motion";

interface VideoPlayerProps {
  type: 'youtube' | 'streamable' | 'direct' | 'embed';
  src: string;
  title: string;
}

export const VideoPlayer = ({ type, src, title }: VideoPlayerProps) => {
  return (
    <div className="relative w-full aspect-video rounded-3xl overflow-hidden shadow-2xl border border-white/10 group">
      {/* Ambient Light Effect */}
      <div className="absolute inset-0 z-[-1] blur-[100px] opacity-20 scale-125 bg-gradient-to-br from-indigo-500 via-purple-500 to-pink-500 animate-pulse"></div>
      
      {type === 'direct' ? (
        <video 
          src={src} 
          controls 
          autoPlay 
          className="w-full h-full object-contain bg-black"
        >
          해당 브라우저는 비디오 태그를 지원하지 않습니다.
        </video>
      ) : (
        <iframe
          src={src}
          title={title}
          allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
          allowFullScreen
          className="w-full h-full border-0 bg-black"
        />
      )}
      
      {/* Title Overlay */}
      <div className="absolute top-0 left-0 w-full p-8 bg-gradient-to-b from-black/80 to-transparent opacity-0 group-hover:opacity-100 transition-opacity">
        <h3 className="text-white font-bold text-lg">{title}</h3>
      </div>
    </div>
  );
};
```

### [Frontend] web/app/video/page.tsx
```tsx
"use client";

import React, { useState } from "react";
import { VideoPlayer } from "@/components/VideoPlayer";
import { Search, ArrowLeft, Play, AlertCircle } from "lucide-react";
import Link from "next/link";

export default function VideoViewerPage() {
  const [url, setUrl] = useState("");
  const [videoData, setVideoData] = useState<any>(null);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState("");

  const handleExtract = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!url) return;

    setIsLoading(true);
    setError("");
    setVideoData(null);

    try {
      const res = await fetch(`http://localhost:8787/api/video-extract?url=${encodeURIComponent(url)}`);
      if (res.ok) {
        const data = await res.json();
        setVideoData(data);
      } else {
        setError("영상 정보를 찾을 수 없습니다. URL을 확인해 주세요.");
      }
    } catch (err) {
      setError("서버 연결에 실패했습니다.");
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-[#09090b] text-white p-6 md:p-12">
      <div className="max-w-5xl mx-auto">
        <div className="flex items-center justify-between mb-12">
          <Link href="/list" className="p-3 hover:bg-white/5 rounded-full transition-colors">
            <ArrowLeft size={24} />
          </Link>
          <h1 className="text-2xl font-black tracking-tighter text-gradient">VIDEO VIEWER MODE</h1>
          <div className="w-12"></div>
        </div>

        {!videoData ? (
          <div className="flex flex-col items-center justify-center py-24">
            <div className="w-20 h-20 bg-indigo-500/20 rounded-3xl flex items-center justify-center text-indigo-400 mb-8 animate-bounce">
              <Play size={40} fill="currentColor" />
            </div>
            <h2 className="text-3xl font-bold mb-4">어떤 영상을 감상할까요?</h2>
            <p className="text-zinc-500 mb-12 text-center max-w-md">
              유튜브, 스트리머블 또는 커뮤니티 게시글 URL을 입력하세요.<br/>광고 없이 깨끗하게 영상만 추출해 드립니다.
            </p>
            
            <form onSubmit={handleExtract} className="w-full max-w-2xl relative">
              <input 
                type="text" 
                value={url}
                onChange={(e) => setUrl(e.target.value)}
                placeholder="https://..."
                className="w-full bg-surface/50 border border-white/10 rounded-2xl py-5 px-6 pl-14 focus:outline-none focus:ring-2 focus:ring-indigo-500/50 transition-all text-lg"
              />
              <Search className="absolute left-6 top-1/2 -translate-y-1/2 text-zinc-500" size={24} />
              <button 
                type="submit"
                disabled={isLoading}
                className="absolute right-3 top-1/2 -translate-y-1/2 px-6 py-3 bg-indigo-500 hover:bg-indigo-600 disabled:opacity-50 text-white rounded-xl font-bold transition-all"
              >
                {isLoading ? "추출 중..." : "START"}
              </button>
            </form>

            <div className="mt-8 flex gap-4 text-xs font-bold text-zinc-600 uppercase tracking-widest">
              <span>Support: Youtube</span>
              <span>•</span>
              <span>Streamable</span>
              <span>•</span>
              <span>Community Links</span>
            </div>
          </div>
        ) : (
          <div className="space-y-8 animate-in fade-in zoom-in duration-500">
            <VideoPlayer {...videoData} />
            
            <div className="flex items-center justify-between">
              <div>
                <span className="text-xs font-bold text-indigo-400 uppercase tracking-widest mb-1 block">Mode: {videoData.type}</span>
                <h3 className="text-xl font-bold">{videoData.title}</h3>
              </div>
              <button 
                onClick={() => setVideoData(null)}
                className="px-6 py-3 bg-white/5 hover:bg-white/10 border border-white/10 rounded-xl text-sm font-bold transition-all"
              >
                다른 영상 보기
              </button>
            </div>
          </div>
        )}

        {error && (
          <div className="mt-8 p-4 bg-red-500/10 border border-red-500/20 rounded-2xl flex items-center gap-3 text-red-400 text-sm animate-pulse">
            <AlertCircle size={18} />
            {error}
          </div>
        )}
      </div>
    </div>
  );
}
```
